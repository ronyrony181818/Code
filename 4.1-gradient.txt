#local minima of a function using gradient descent
x = 2
lr = 0.01
precision=0.000001
previous_step_size=1
iters=0
max_iters=10000
gf= lambda x : (x+3)**2

import matplotlib.pyplot as plt
gd=[]

while precision < previous_step_size and iters < max_iters:
  prev=x
  x=x-(lr*gf(x))
  previous_step_size=abs(x-prev)
  iters+=1
  gd.append(x)

print("Iteration",iters,"\nX value is",x)

print("Local Minima: ", x)

plt.plot(range(len(gd)) , gd)
